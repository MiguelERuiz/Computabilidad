\section*{Ejercicio 8.9}
\subsection*{Solución}
\begin{lstlisting}[language=pascal]
  proc p(n:int) retunrs int{
    if (n+4 > 10 ^ 4-n < 1) then{
      return 0;
    }
    if (n+4 > 10) then{
      return n+8;
    }
    var i:int, j:int;
    i:=0;
    j:=0;
    while(i<6) do{
      j:=j+n;
      i:=i+1;
    }
    return j;
  }
\end{lstlisting}
