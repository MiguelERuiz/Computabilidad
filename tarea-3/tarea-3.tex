\section*{Ejercicio 5.2}
\subsection*{Solución}
\begin{lstlisting}[language=pascal]
  proc fibonacci(x : int) returns int{
    if (x<=1) then return x;
    var res: int;
    var resAnterior: int;
    var i: int;
    var n: int;
    n = x-1;
    res = 1;
    for i = 2 to n do {
      var temp: int;
      temp = res;
      res = res + resAnterior;
      resAnterior = temp;
    }
    return res;
  }
\end{lstlisting}
